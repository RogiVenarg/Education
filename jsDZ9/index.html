class CssClass {
  #myStyle = [];
  constructor (name){
    this.name = name;
  }
  pushMyStyle = function(name, val){
    let str = name + ": " + val + ";\n"
    this.#myStyle.push(str);
  }
  delMyStyle = function(ind){
    ind = Number(ind);
    if ((isNaN(ind)) || (ind < 0) || (ind > this.#myStyle.length - 1)) return undefined;
    else this.#myStyle.splice(ind, 1);
  }
  getCss = function(){
    let str = "."
    let i;
    str += this.name + "{\n";
    for (i = 0; i < this.#myStyle.length; i++){str += this.#myStyle[i]}
    str += "}";
    return str;
  }
}
